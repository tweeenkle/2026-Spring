  <h3>코디 조건</h3>

  <select id="weather">
    <option value="hot">더움 (25도+)</option>
    <option value="mild">선선 (15~25도)</option>
    <option value="cold">추움 (15도 이하)</option>
  </select>

    <option>봄</option>
    <option>여름</option>
    <option>가을</option>
    <option>겨울</option>
  </select>

  <button onclick="recommend()">코디 추천</button>

<script>
let closet = JSON.parse(localStorage.getItem('closet')||'[]');
const closetEl = document.getElementById('closet');

function save(){ localStorage.setItem('closet', JSON.stringify(closet)); }

function render(){
  closetEl.innerHTML='';
  closet.forEach(i=>{
    const div=document.createElement('div');
    div.innerText=`${i.name} (${i.season}/${i.thickness})`;
    closetEl.appendChild(div);
  });
}

function addItem(){
  const item={
    category:category.value,
    season:season.value,
    thickness:thickness.value,
    name:name.value
  };
  closet.push(item);
  save(); render();
}

function filterItems(cat, season, weather){
  return closet.filter(i=>{
    if(i.category!==cat) return false;
    if(i.season!==season) return false;

    if(weather==='hot' && i.thickness==='thick') return false;
    if(weather==='cold' && i.thickness==='light') return false;

    return true;
  });
}

function pick(arr){ return arr[Math.floor(Math.random()*arr.length)]; }

function recommend(){
  const weather = document.getElementById('weather').value;
  const season = document.getElementById('seasonSelect').value;

  const tops = filterItems('top',season,weather);
  const bottoms = filterItems('bottom',season,weather);
  const outers = filterItems('outer',season,weather);
  const shoes = filterItems('shoes',season,weather);

  const result = document.getElementById('result');
  result.innerHTML='';

  if(!tops.length||!bottoms.length||!shoes.length){
    result.innerHTML='옷이 부족해!'; return;
  }

  const top = pick(tops);
  const bottom = pick(bottoms);
  const shoe = pick(shoes);
  const outer = (weather!=='hot' && outers.length)? pick(outers):null;

  result.innerHTML=`
    <div class='outfit'>
      👕 ${top.name}<br>
      👖 ${bottom.name}<br>
      ${outer?`🧥 ${outer.name}<br>`:''}
      👟 ${shoe.name}
    </div>
  `;
}

render();
</script>
