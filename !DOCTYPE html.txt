  <input type="text" id="input" placeholder="편하게 이야기해 주세요...">
  <button onclick="sendMessage()">전송</button>
</div>

<script>
  const chatContainer = document.getElementById('chat-container');

  function addMessage(text, sender) {
    const message = document.createElement('div');
    message.classList.add('message', sender);
    message.innerText = text;
    chatContainer.appendChild(message);
    chatContainer.scrollTop = chatContainer.scrollHeight;
  }

  function showTyping() {
    const typing = document.createElement('div');
    typing.classList.add('message', 'bot', 'typing');
    typing.innerText = '상담사가 생각 중이에요...';
    typing.id = 'typing';
    chatContainer.appendChild(typing);
    chatContainer.scrollTop = chatContainer.scrollHeight;
  }

  function removeTyping() {
    const typing = document.getElementById('typing');
    if (typing) typing.remove();
  }

  function sendMessage() {
    const input = document.getElementById('input');
    const text = input.value.trim();
    if (!text) return;

    addMessage(text, 'user');
    input.value = '';

    showTyping();

    setTimeout(() => {
      removeTyping();
      const response = generateResponse(text);
      addMessage(response, 'bot');
    }, 1000);
  }

  function quickMessage(text) {
    document.getElementById('input').value = text;
    sendMessage();
  }

  function generateResponse(userText) {
    const lower = userText.toLowerCase();

    if (lower.includes('힘들')) {
      return '많이 버티고 있었구나. 요즘 어떤 일이 가장 크게 힘들게 해?';
    }

    if (lower.includes('불안')) {
      return '불안할 때는 생각이 계속 커지기 쉬워. 지금 머릿속에 가장 큰 걱정은 뭐야?';
    }

    if (lower.includes('외로')) {
      return '외롭다는 느낌은 정말 무겁지. 요즘 혼자라고 느껴지는 순간이 언제야?';
    }

    if (lower.includes('괜찮')) {
      return '괜찮아지고 싶다는 마음이 이미 중요한 시작이야. 지금 나아지기 위해 해보고 싶은 게 있을까?';
    }

    return '네 이야기를 더 듣고 싶어. 조금 더 자세히 말해줄 수 있을까?';
  }

  document.getElementById('input').addEventListener('keypress', function(e) {
    if (e.key === 'Enter') sendMessage();
  });

  // 초기 메시지
  setTimeout(() => {
    addMessage('안녕, 나는 너의 이야기를 들어주는 상담 챗봇이야. 지금 어떤 기분이야?', 'bot');
  }, 500);
</script>
